Оператор Описание
= Равно
< Меньше чем
> Больше чем
<= Меньше или равно
>= Больше или равно
<>, != Не равно 
SELECT order_id, item, amount
    FROM Orders
    WHERE customer_id = 4;
SELECT item, amount, amount-20 AS offer_price
    FROM Orders;
SELECT order_id, item, amount
    FROM Orders
    WHERE amount < 400;
SELECT order_id, item, amount
    FROM Orders
    WHERE amount > 400;
SELECT order_id, item, amount
    FROM Orders
    WHERE amount <= 400;
SELECT order_id, item, amount
    FROM Orders
    WHERE amount >= 400;
SELECT order_id, item, amount
    FROM Orders
    WHERE amount != 400;
